function getKeyAndMove(e){				
    var key_code=e.which||e.keyCode;
        switch(key_code){
            case 37: left();
				break;
            case 38: up();
				break;
            case 39: right();
                break;
            case 40: down();
				break;
                }}


// Fonctions déplacement // 

function right() {
    var pion=document.getElementById('pion');
    pion.style.transform='translateX(40px)';
}

function left() {
    var pion=document.getElementById('pion');
    pion.style.transform='translateX(-40px)';
}

function down() {
    var pion=document.getElementById('pion');
    pion.style.transform='translateY(40px)';   
}

function up() {
    var pion=document.getElementById('pion');
    pion.style.transform='translateY(-40px)';
}
